   - Line count > 300 lines

   ```bash

   complexity_trigger=false

     if [[ -f "$file" ]]; then
       line_count=$(wc -l < "$file" 2>/dev/null || echo "0")
       if [[ "$line_count" -gt 300 ]]; then
         complexity_trigger=true
         files_needing_refactor="$files_needing_refactor $file"
         continue
       fi

         complexity_trigger=true
         files_needing_refactor="$files_needing_refactor $file"
   ```
