  local dir="$1"
  local questions_json="$2"
  local sources_json="${3:-{}}"

{
  "version": "1.1",
  "sources": $sources_json,
  "questions": $questions_json,
  "created_at": "2026-03-13T00:00:00Z",
  "last_updated": "2026-03-13T00:00:00Z"
}
EOF
}
