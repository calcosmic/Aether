```go
package learn

// Classification enum (D-10, D-11)
type Classification string

const (
    ClassBlocked       Classification = "blocked"
    ClassRepoLocal     Classification = "repo-local"
    ClassHiveShareable Classification = "hive-shareable"
    ClassNeedsApproval Classification = "needs-user-approval"
)

// WorkerEvidence records a single worker's contribution (D-09)
type WorkerEvidence struct {
    Name   string `json:"name"`
    Caste  string `json:"caste"`
    Status string `json:"status"`
}

// Evidence carries full structured provenance for a learning entry (D-09)
type Evidence struct {
    RunID        string           `json:"run_id"`
    Phase        int              `json:"phase"`
    Workers      []WorkerEvidence `json:"workers"`
    FilesTouched []string         `json:"files_touched,omitempty"`
    GatesPassed  int              `json:"gates_passed"`
    GatesTotal   int              `json:"gates_total"`
    Confidence   float64          `json:"confidence"`
    Timestamp    string           `json:"timestamp"`
    Scope        string           `json:"scope"`
}

// Entry is a single durable learning record
type Entry struct {
    ID             string        `json:"id"`
    Content        string        `json:"content"`
    Evidence       Evidence      `json:"evidence"`
    Classification Classification `json:"classification"`
    CreatedAt      string        `json:"created_at"`
    Phase          int           `json:"phase"`
    Caste          string        `json:"caste,omitempty"`
    FilePath       string        `json:"file_path,omitempty"`
    Confidence     float64       `json:"confidence"`
    Redacted       bool          `json:"redacted,omitempty"`
}

// EntryFilter for List queries
type EntryFilter struct {
    Phase          int            `json:"phase,omitempty"`
    Classification Classification `json:"classification,omitempty"`
    MinConfidence  float64        `json:"min_confidence,omitempty"`
    Limit          int            `json:"limit,omitempty"`
}

// LearnStore interface (D-07) -- ColonyStore and HiveStore implement this
type LearnStore interface {
    Add(entry Entry) error
    Get(id string) (*Entry, error)
    List(filter EntryFilter) ([]Entry, error)
    Replace(id string, entry Entry) error
    Remove(id string) error
    Compact(budget int) error
}
```

GREEN phase: Create pkg/learn/colony_store.go implementing ColonyStore:
- Struct holds `*storage.Store` and the relative file path `"entries.json"` (D-06: data in .aether/data/learn/)
- `NewColonyStore(store *storage.Store) *ColonyStore` constructor
- `Add`: generate ID as `fmt.Sprintf("lrn_%s_%d", time.Now().Format("20060102"), len(entries))` or similar. Use `store.UpdateJSONAtomically` to read-modify-write the entries slice. Append entry. Write back.
- `Get`: load entries, find by ID, return pointer or nil
- `List`: load entries, apply EntryFilter (phase match, classification match, min_confidence, limit). Return empty slice (not nil) when no matches.
- `Replace`: load entries, find by ID, update in place, write back. Return error if not found.
- `Remove`: load entries, filter out by ID, write back. Return error if not found.
- `Compact(budget int)`: sort entries by Confidence descending (highest first). Keep entries until cumulative content length exceeds budget. Remove the rest. Write back.
- All load operations: if file does not exist, return empty slice / nil / not-found (no errors for missing files).

Follow conventions from pkg/events/bus.go: accept *storage.Store in constructor, use store.LoadJSON/SaveJSON, return empty slices not nil, use UpdateJSONAtomically for read-modify-write.
