    type depEntry struct {
        Name    string `json:"name"`
        Version string `json:"version,omitempty"`
    }

    // techStackDetail holds parsed dependency data for a single language ecosystem.
    type techStackDetail struct {
        Language      string    `json:"language"`
        SourceFile    string    `json:"source_file"`
        Dependencies  []depEntry `json:"dependencies"`
        DevDeps       []depEntry `json:"dev_dependencies,omitempty"`
        Indirect      []depEntry `json:"indirect,omitempty"`
    }

    **3. Implement parser functions** -- add each as a standalone function following the tolerant parsing pattern (return nil/empty on error, never panic):

    **parsePackageJsonDeps(target string) ([]depEntry, []depEntry):**
    - Read `filepath.Join(target, "package.json")`
    - Use `gjson.GetBytes(data, "dependencies")` to iterate the `dependencies` map. For each key-value, create a `depEntry{Name: key, Version: value.String()}`
    - Use `gjson.GetBytes(data, "devDependencies")` for dev deps
    - Return `prodDeps, devDeps`

    **parseGoModDeps(target string) ([]depEntry, []depEntry):**
    - Read `filepath.Join(target, "go.mod")`
    - Split into lines. Track `inRequireBlock` for `require (` ... `)` blocks
    - For each line with a module path + version: if line contains `// indirect`, add to indirect slice, else add to direct slice
    - Also handle single-line `require module/path v1.2.3` syntax
    - Return `direct, indirect`

    **parseCargoTomlDeps(target string) []depEntry:**
    - Read `filepath.Join(target, "Cargo.toml")`
    - Parse with `toml.Decode()` into `map[string]interface{}`
    - Extract `deps["dependencies"]` map, iterate keys to create depEntry entries
    - Values may be strings (version) or maps (table with version key) -- handle both
    - Return deps

    **parsePyprojectDeps(target string) []depEntry:**
    - Read `filepath.Join(target, "pyproject.toml")`
    - Parse with `toml.Decode()` into `map[string]interface{}`
    - Check `project["dependencies"]` first (PEP 621)
    - Also check `toolMap["poetry"]["dependencies"]` as fallback
    - Each entry may be a string like `"requests>=2.0"` -- split on version operators to extract name/version
    - Return deps

    **parseRequirementsTxt(target string) []depEntry:**
    - Read `filepath.Join(target, "requirements.txt")`
    - Split into lines, skip blank lines and lines starting with `#`, `-r`, `-e`, `--`
    - For each dep line: split on first occurrence of `==`, `>=`, `<=`, `~=`, `!=`, `>`, `<` to get name and version
    - Return deps

    **parseGemfileDeps(target string) []depEntry:**
    - Read `filepath.Join(target, "Gemfile")`
    - Use `regexp.MustCompile(`gem\s+["']([^"']+)["']\s*(?:,\s*["']([^"']+)["'])?`)` to find gem entries
    - For each match: group 1 = name, group 2 = version (may be empty)
    - Return deps

    **parsePomXmlDeps(target string) []depEntry:**
    - Read `filepath.Join(target, "pom.xml")`
    - Define minimal XML structs:
      ```go
      type mavenProject struct {
          XMLName    xml.Name `xml:"project"`
          Dependencies struct {
              Deps []mavenDep `xml:"dependency"`
          } `xml:"dependencies"`
      }
      type mavenDep struct {
          GroupID    string `xml:"groupId"`
          ArtifactID string `xml:"artifactId"`
          Version    string `xml:"version"`
      }
      ```
    - Parse with `xml.Unmarshal()`, build depEntry from ArtifactID + Version
    - Return deps

    **parseMixExsDeps(target string) []depEntry:**
    - Read `filepath.Join(target, "mix.exs")`
    - Find content between `defp deps do` and `end` block
    - Use `regexp.MustCompile(`\{:(\w+)`)` to extract atom names from `{:name, ...}` tuples
    - Return deps

    **parseComposerJsonDeps(target string) ([]depEntry, []depEntry):**
    - Read `filepath.Join(target, "composer.json")`
    - Use `gjson.GetBytes(data, "require")` for production deps
    - Use `gjson.GetBytes(data, "require-dev")` for dev deps
    - Same gjson map iteration as package.json
    - Return `prodDeps, devDeps`

    **4. Add orchestrating function:**
    ```go
    func parseDependencyFiles(target string) []techStackDetail {
        var details []techStackDetail

        if hasFile(target, "package.json") {
            prod, dev := parsePackageJsonDeps(target)
            details = append(details, techStackDetail{
                Language: "node", SourceFile: "package.json",
                Dependencies: prod, DevDeps: dev,
            })
        }
        if hasFile(target, "go.mod") {
            direct, indirect := parseGoModDeps(target)
            details = append(details, techStackDetail{
                Language: "go", SourceFile: "go.mod",
                Dependencies: direct, Indirect: indirect,
            })
        }
        if hasFile(target, "Cargo.toml") {
            deps := parseCargoTomlDeps(target)
            details = append(details, techStackDetail{
                Language: "rust", SourceFile: "Cargo.toml",
                Dependencies: deps,
            })
        }
        if hasFile(target, "pyproject.toml") {
            deps := parsePyprojectDeps(target)
            details = append(details, techStackDetail{
                Language: "python", SourceFile: "pyproject.toml",
                Dependencies: deps,
            })
        }
        if hasFile(target, "requirements.txt") {
            deps := parseRequirementsTxt(target)
            details = append(details, techStackDetail{
                Language: "python", SourceFile: "requirements.txt",
                Dependencies: deps,
            })
        }
        if hasFile(target, "Gemfile") {
            deps := parseGemfileDeps(target)
            details = append(details, techStackDetail{
                Language: "ruby", SourceFile: "Gemfile",
                Dependencies: deps,
            })
        }
        if hasFile(target, "pom.xml") {
            deps := parsePomXmlDeps(target)
            details = append(details, techStackDetail{
                Language: "java", SourceFile: "pom.xml",
                Dependencies: deps,
            })
        }
        if hasFile(target, "mix.exs") {
            deps := parseMixExsDeps(target)
            details = append(details, techStackDetail{
                Language: "elixir", SourceFile: "mix.exs",
                Dependencies: deps,
            })
        }
        if hasFile(target, "composer.json") {
            prod, dev := parseComposerJsonDeps(target)
            details = append(details, techStackDetail{
                Language: "php", SourceFile: "composer.json",
                Dependencies: prod, DevDeps: dev,
            })
        }

        return details
    }
    ```
