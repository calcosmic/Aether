    **Part A: Create systematic audit test (resolves RESEARCH.md open questions)**

    Create `cmd/cli_flag_audit_test.go` with a test that:
    1. Reads all `.claude/commands/ant/*.md` and `.aether/docs/command-playbooks/*.md` files
    2. Extracts every `aether <subcommand> --<flag> <value>` pattern from those files
    3. For each unique subcommand+flag combination, verifies the flag is registered in Go runtime via `rootCmd.Commands()` lookup and `cmd.Flags().Lookup(flag)`

    Implementation approach:
    ```go
    package cmd

    import (
        "os"
        "regexp"
        "strings"
        "testing"
    )

    // TestCLIFlagAudit systematically compares markdown CLI calls against Go registrations.
    // This resolves RESEARCH.md open questions 1 and 2 with concrete evidence.
    func TestCLIFlagAudit(t *testing.T) {
        // Directories to scan for markdown CLI calls
        markdownDirs := []string{
            "../.claude/commands/ant/",
            "../.opencode/commands/ant/",
            "../.aether/docs/command-playbooks/",
        }

        // Regex to extract "aether <subcommand>" calls with flags
        // Matches: aether subcommand-name --flag-name value
        re := regexp.MustCompile(`aether\s+([\w][\w-]*)\s+((?:--[\w][\w-]*(?:=\S*|\s+\S*)?\s*)*)`)

        // Build lookup: subcommand name -> set of registered flags
        registered := make(map[string]map[string]bool)
        for _, cmd := range rootCmd.Commands() {
            flags := make(map[string]bool)
            cmd.Flags().VisitAll(func(f) {
                flags[f.Name] = true
            })
            // Also include persistent flags
            cmd.PersistentFlags().VisitAll(func(f) {
                flags[f.Name] = true
            })
            registered[cmd.Name()] = flags
        }

        var mismatches []string

        for _, dir := range markdownDirs {
            entries, err := os.ReadDir(dir)
            if err != nil {
                continue // directory may not exist in test environment
            }
            for _, entry := range entries {
                if !strings.HasSuffix(entry.Name(), ".md") {
                    continue
                }
                data, err := os.ReadFile(dir + entry.Name())
                if err != nil {
                    continue
                }
                matches := re.FindAllStringSubmatch(string(data), -1)
                for _, m := range matches {
                    subcmd := m[1]
                    flagsStr := m[2]

                    // Check subcommand exists
                    flagSet, exists := registered[subcmd]
                    if !exists {
                        mismatches = append(mismatches, fmt.Sprintf(
                            "%s%s: subcommand %q not registered in Go runtime",
                            dir, entry.Name(), subcmd))
                        continue
                    }

                    // Extract --flag names from the flags portion
                    flagRe := regexp.MustCompile(`--([\w][\w-]*)`)
                    flagMatches := flagRe.FindAllStringSubmatch(flagsStr, -1)
                    for _, fm := range flagMatches {
                        flagName := fm[1]
                        if !flagSet[flagName] {
                            mismatches = append(mismatches, fmt.Sprintf(
                                "%s%s: subcommand %q missing flag --%s",
                                dir, entry.Name(), subcmd, flagName))
                        }
                    }
                }
            }
        }

        if len(mismatches) > 0 {
            t.Errorf("CLI flag audit found %d mismatches:\n%s",
                len(mismatches), strings.Join(mismatches, "\n"))
        }
    }
    ```

    IMPORTANT: This test MUST be written to pass initially. The codebase evidence shows most flags already exist (pheromone-write has all 6 flags, midden-write has all 3, memory-capture has both, spawn-log has all 7, activity-log has all 4).     package cmd

    ```go
    import (
        "bytes"
        "strings"
        "testing"
    )

    // TestSubcommandSmokeTest verifies that all registered subcommands
    // respond cleanly to --help. This catches missing flags, panics,
    // and broken registrations. Covers PLAT-05.
    func TestSubcommandSmokeTest(t *testing.T) {
        commands := rootCmd.Commands()

        for _, cmd := range commands {
            cmd := cmd
            t.Run(cmd.Name(), func(t *testing.T) {
                saveGlobals(t)
                resetRootCmd(t)

                var buf bytes.Buffer
                stdout = &buf
                stderr = &buf

                rootCmd.SetArgs([]string{cmd.Name(), "--help"})
                err := rootCmd.Execute()

                if err != nil {
                    t.Errorf("subcommand %q --help returned error: %v", cmd.Name(), err)
                }
                if buf.Len() == 0 {
                    t.Errorf("subcommand %q --help produced no output", cmd.Name())
                }
            })
        }
    }
    ```
