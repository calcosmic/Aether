func attachExternalContinueWatcher(verification codexContinueVerificationReport, workerFlow []codexContinueWorkerFlowStep) (codexContinueVerificationReport, *codexContinueWorkerFlowStep) {
	for _, step := range workerFlow {
		if strings.TrimSpace(step.Stage) != "verification" || !strings.EqualFold(strings.TrimSpace(step.Caste), "watcher") {
			continue
		}
		status := continueWorkerFlowStatus(step.Status)
		summary := strings.TrimSpace(step.Summary)
		if summary == "" {
			summary = continueWatcherDefaultSummary(status)
		}
		watcher := codexWatcherVerification{
			Present: true,
			Passed:  status == "completed" || status == "manually-reconciled",
			Status:  status,
			Worker:  strings.TrimSpace(step.Name),
			Summary: summary,
		}
		verification.Watcher = watcher
		if !watcher.Passed {
			verification.ChecksPassed = false
			verification.Passed = false
			verification.BlockingIssues = uniqueSortedStrings(append(verification.BlockingIssues, summary))
		}
		watcherFlow := step
		watcherFlow.Summary = continueWatcherFlowSummary(watcher.Worker, watcher.Status, watcher.Summary)
		return verification, &watcherFlow
	}
	verification.ChecksPassed = false
	verification.Passed = false
	verification.BlockingIssues = uniqueSortedStrings(append(verification.BlockingIssues, "wrapper continue watcher result is missing"))
	return verification, nil
}

## Decisions from CONTEXT.md

- **D-01:** When the external continue watcher times out (status = "timeout") but all runtime verification steps passed, advancement is ALLOWED with a warning. The timeout is treated as advisory, not blocking.
- **D-02:** When the external continue watcher fails (status = "failed"), advancement remains BLOCKED. Only timeout status gets the advisory treatment.
- **D-03:** The warning is recorded in `BlockingIssues` for visibility but does not set `Passed = false` or `ChecksPassed = false` when status is "timeout" and verification steps passed.
- **D-04:** Existing tests `TestContinue_BlocksOnWatcherTimeout` and `TestContinue_BlocksOnWatcherTimeout_Integration` assert the OLD behavior (blocking on timeout). These test the INTERNAL watcher path (`runCodexContinueVerification`), not the EXTERNAL watcher path (`attachExternalContinueWatcher`). The internal watcher path behavior remains unchanged -- those tests should continue to pass.
- **D-05:** The failing test `TestAttachExternalContinueWatcherTimeoutAdvisory` defines the target behavior for the EXTERNAL watcher path. This test must pass after the fix.
- **D-06:** This is a surgical fix to `attachExternalContinueWatcher()` only. No changes to `runCodexContinueVerification()`, `runCodexContinueWatcherVerification()`, or other continue logic.
