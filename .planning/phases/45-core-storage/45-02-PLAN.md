       ```go
       package storage

       import (
           "fmt"
           "os"
           "path/filepath"
           "sort"
           "time"
       )

       const maxBackups = 3

       // CreateBackup copies the file at path to path + ".bak.{timestamp}" if the original exists.
       // Returns nil if original does not exist (nothing to back up).
       func (s *Store) CreateBackup(path string) error {
           if _, err := os.Stat(path); os.IsNotExist(err) {
               return nil // nothing to back up
           }
           backupPath := fmt.Sprintf("%s.bak.%s", path, time.Now().Format("20060102-150405"))
           data, err := os.ReadFile(path)
           if err != nil {
               return fmt.Errorf("storage: read for backup %q: %w", path, err)
           }
           if err := os.WriteFile(backupPath, data, 0644); err != nil {
               return fmt.Errorf("storage: write backup %q: %w", backupPath, err)
           }
           return nil
       }

       // RotateBackups removes the oldest backup files beyond maxBackups.
       // Backup files are identified by the pattern: path + ".bak.*"
       func (s *Store) RotateBackups(basePath string) error {
           pattern := basePath + ".bak.*"
           matches, err := filepath.Glob(pattern)
           if err != nil {
               return fmt.Errorf("storage: glob backups for %q: %w", basePath, err)
           }
           if len(matches) <= maxBackups {
               return nil
           }
           // Sort by modification time, newest first
           sort.Slice(matches, func(i, j int) bool {
               si, _ := os.Stat(matches[i])
               sj, _ := os.Stat(matches[j])
               return si.ModTime().After(sj.ModTime())
           })
           // Delete oldest beyond maxBackups
           for _, f := range matches[maxBackups:] {
               os.Remove(f)
           }
           return nil
       }
       ```

    2. CREATE pkg/storage/paths.go (per D-07: AETHER_ROOT -> git root -> cwd fallback):

       ```go
       package storage

       import (
           "os"
           "os/exec"
           "path/filepath"
           "strings"
       )

       // ResolveAetherRoot resolves the Aether root directory using a 3-tier fallback:
       // 1. AETHER_ROOT environment variable (if set)
       // 2. Git repository root (via git rev-parse --show-toplevel)
       // 3. Current working directory (fallback)
       func ResolveAetherRoot() string {
           if root := os.Getenv("AETHER_ROOT"); root != "" {
               return root
           }
           // Try git root
           cmd := exec.Command("git", "rev-parse", "--show-toplevel")
           if out, err := cmd.Output(); err == nil {
               return strings.TrimSpace(string(out))
           }
           // Fallback to cwd
           dir, _ := os.Getwd()
           return dir
       }

       // ResolveDataDir resolves the colony data directory.
       // If COLONY_DATA_DIR is set, use it directly.
       // Otherwise, use AETHER_ROOT/.aether/data/.
       func ResolveDataDir() string {
           if dir := os.Getenv("COLONY_DATA_DIR"); dir != "" {
               return dir
           }
           return filepath.Join(ResolveAetherRoot(), ".aether", "data")
       }
       ```
